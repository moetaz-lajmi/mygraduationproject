// src/pages/Vendre.jsx
import React, { useState } from "react";

const Vendre = () => {
  const [formData, setFormData] = useState({
    titre: "",
    prix: "",
    pieces: "1",
    chambres: "1",
    surface: "",
    adresse: "",
    description: "",
    photos: [],
  });

  const handleChange = (e) => {
    const { name, value } = e.target;
    setFormData((prev) => ({ ...prev, [name]: value }));
  };

  const handleFileChange = (e) => {
    const files = Array.from(e.target.files);
    setFormData((prev) => ({ ...prev, photos: files }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    console.log("Annonce publiée :", formData);
    alert("Annonce publiée avec succès !");
  };

  return (
    <div className="min-h-screen bg-gray-50">
      {/* Section bannière */}
      <section className="bg-emerald-700 text-white py-20 text-center">
        <h1 className="text-4xl font-bold mb-4">Vendez votre maison facilement</h1>
        <p className="text-lg mb-6">
          Publiez votre annonce en quelques clics et trouvez l’acheteur idéal pour votre propriété
        </p>
        <a
          href="#formulaire"
          className="bg-white text-emerald-700 px-6 py-3 rounded-lg font-semibold hover:bg-gray-100"
        >
          + Publier une annonce
        </a>
      </section>

      {/* Étapes */}
      <section className="py-16 px-6 max-w-5xl mx-auto text-center">
        <h2 className="text-2xl font-bold mb-10">Comment vendre avec MaisonConnect</h2>
        <div className="grid grid-cols-1 md:grid-cols-3 gap-8">
          <div className="bg-white shadow rounded-2xl p-6">
            <h3 className="font-semibold mb-2 text-emerald-700">1. Créez votre annonce</h3>
            <p>Ajoutez photos, description détaillée et prix de votre propriété.</p>
          </div>
          <div className="bg-white shadow rounded-2xl p-6">
            <h3 className="font-semibold mb-2 text-emerald-700">2. Recevez des contacts</h3>
            <p>Les acheteurs intéressés vous contactent directement via la plateforme.</p>
          </div>
          <div className="bg-white shadow rounded-2xl p-6">
            <h3 className="font-semibold mb-2 text-emerald-700">3. Finalisez la vente</h3>
            <p>Organisez des visites et négociez directement avec les acheteurs.</p>
          </div>
        </div>
      </section>

      {/* Formulaire de publication */}
      <section id="formulaire" className="py-16 bg-white shadow-inner">
        <div className="max-w-3xl mx-auto px-6">
          <h2 className="text-2xl font-bold text-center mb-8">Publier votre annonce</h2>
          <form onSubmit={handleSubmit} className="space-y-6">
            {/* Titre & Prix */}
            <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
              <div>
                <label className="block mb-1 font-semibold">Titre de l’annonce</label>
                <input
                  type="text"
                  name="titre"
                  value={formData.titre}
                  onChange={handleChange}
                  placeholder="Ex: Maison familiale avec jardin"
                  className="w-full border rounded-lg p-2"
                  required
                />
              </div>
              <div>
                <label className="block mb-1 font-semibold">Prix (€)</label>
                <input
                  type="number"
                  name="prix"
                  value={formData.prix}
                  onChange={handleChange}
                  placeholder="250000"
                  className="w-full border rounded-lg p-2"
                  required
                />
              </div>
            </div>

            {/* Pièces, chambres, surface */}
            <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
              <div>
                <label className="block mb-1 font-semibold">Pièces</label>
                <select
                  name="pieces"
                  value={formData.pieces}
                  onChange={handleChange}
                  className="w-full border rounded-lg p-2"
                >
                  {[1, 2, 3, 4, 5].map((n) => (
                    <option key={n} value={n}>{n}</option>
                  ))}
                </select>
              </div>
              <div>
                <label className="block mb-1 font-semibold">Chambres</label>
                <select
                  name="chambres"
                  value={formData.chambres}
                  onChange={handleChange}
                  className="w-full border rounded-lg p-2"
                >
                  {[1, 2, 3, 4, 5].map((n) => (
                    <option key={n} value={n}>{n}</option>
                  ))}
                </select>
              </div>
              <div>
                <label className="block mb-1 font-semibold">Surface (m²)</label>
                <input
                  type="number"
                  name="surface"
                  value={formData.surface}
                  onChange={handleChange}
                  placeholder="120"
                  className="w-full border rounded-lg p-2"
                  required
                />
              </div>
            </div>

            {/* Adresse */}
            <div>
              <label className="block mb-1 font-semibold">Adresse</label>
              <input
                type="text"
                name="adresse"
                value={formData.adresse}
                onChange={handleChange}
                placeholder="123 Rue de la Paix, 75001 Paris"
                className="w-full border rounded-lg p-2"
                required
              />
            </div>

            {/* Description */}
            <div>
              <label className="block mb-1 font-semibold">Description</label>
              <textarea
                name="description"
                value={formData.description}
                onChange={handleChange}
                placeholder="Décrivez votre propriété..."
                className="w-full border rounded-lg p-2 h-32"
                required
              ></textarea>
            </div>

            {/* Photos */}
            <div>
              <label className="block mb-1 font-semibold">Photos</label>
              <div className="border-dashed border-2 border-gray-300 rounded-lg p-6 text-center cursor-pointer hover:border-emerald-500">
                <input
                  type="file"
                  name="photos"
                  accept="image/*"
                  multiple
                  onChange={handleFileChange}
                  className="hidden"
                  id="photoUpload"
                />
                <label htmlFor="photoUpload" className="cursor-pointer text-gray-500">
                  Glissez-déposez vos photos ici ou cliquez pour sélectionner
                </label>
              </div>
            </div>

            {/* Bouton */}
            <div className="text-center">
              <button
                type="submit"
                className="bg-emerald-700 text-white px-8 py-3 rounded-lg font-semibold hover:bg-emerald-800"
              >
                Publier l’annonce
              </button>
            </div>
          </form>
        </div>
      </section>
    </div>
  );
};

export default Vendre;



ROUTES: 
import Vendre from "./pages/Vendre";

function App() {
  return (
    <Routes>
      <Route path="/vendre" element={<Vendre />} />
    </Routes>
  );
}
